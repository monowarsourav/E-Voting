% =====================================================================
%  G4 - Blockchain TPS and latency vs Caliper round size
%  --------------------------------------------------------------------
%  Backs claim: 199.9 TPS sustained at 10K with 100% success;
%  failures at 25K+ are due to Fabric Gateway concurrency limit (500),
%  not chaincode logic.
%  Source: chapter4.tex Table tab:caliper
% =====================================================================
\begin{tikzpicture}
\begin{axis}[
    width=0.85\textwidth,
    height=7cm,
    axis y line*=left,
    xlabel={Caliper round (transactions submitted)},
    ylabel={Throughput (TPS)},
    ymin=0, ymax=320,
    symbolic x coords={10K, 25K, 50K, 100K},
    xtick=data,
    bar width=14pt,
    grid=major,
    grid style={dashed, gray!30},
    legend style={at={(0.02,0.98)}, anchor=north west,
                  font=\sffamily\scriptsize, draw=black},
    label style={font=\sffamily\small},
    tick label style={font=\sffamily\scriptsize},
]
% Throughput as bar chart
\addplot+[ybar, fill=gray!20, draw=black, very thick] coordinates {
    (10K,   199.9)
    (25K,   279.3)
    (50K,   227.8)
    (100K,  216.4)
};
\addlegendentry{Throughput (TPS, left axis)}
\end{axis}

% Right axis for latency (overlay)
\begin{axis}[
    width=0.85\textwidth,
    height=7cm,
    axis y line*=right,
    axis x line=none,
    ylabel={Average latency (s)},
    ymin=0, ymax=5,
    symbolic x coords={10K, 25K, 50K, 100K},
    xtick=data,
    legend style={at={(0.02,0.86)}, anchor=north west,
                  font=\sffamily\scriptsize, draw=black},
    label style={font=\sffamily\small},
    tick label style={font=\sffamily\scriptsize},
    every axis plot/.append style={thick},
]
\addplot+[mark=*, mark size=3pt, color=black, mark options={fill=black},
          dashed, very thick] coordinates {
    (10K,   0.11)
    (25K,   2.96)
    (50K,   3.70)
    (100K,  3.98)
};
\addlegendentry{Average latency (s, right axis)}
\end{axis}
\end{tikzpicture}
