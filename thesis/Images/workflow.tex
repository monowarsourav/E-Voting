% =====================================================================
%  Voter Workflow / Activity Diagram for CovertVote (redesigned)
%  --------------------------------------------------------------------
%  Two swimlanes:
%      Phase B - Voter Registration (left, blue tint)
%      Phase C - Vote Casting       (right, yellow tint)
%  Mirrors the corrected protocol:
%      - SMDC real-slot index is server-keyed:
%          idx_real = HMAC(k_srv, "smdc-real-index:" || vid || eid) mod k
%      - Credential layer carries Pedersen commitments C_i and
%        per-slot binary + sum proofs (issued during registration).
%      - Cast-time uses 1-hot Paillier encryption + CDS OR-proofs
%        (no Pedersen commitment at cast time).
%      - behaviorWeight feeds a per-candidate scalar multiplication
%        E_{w,j} = E_j^{fW} BEFORE the coercion-branch decision.
%  Formal pastel coloring + dashed wrapper labels match the other
%  Chapter 3 figures (architecture, smdc, pipeline).
% =====================================================================
\begin{tikzpicture}[
  font=\sffamily\footnotesize,
  >={Stealth[length=2.0mm,width=1.4mm]},
  every node/.style={align=center},
  %------ node shapes ------
  start/.style    ={circle, draw=black, fill=black, minimum size=5mm, inner sep=0pt},
  activity/.style ={rectangle, draw=black, thick, rounded corners=4pt,
                    minimum width=5.2cm, minimum height=0.95cm, fill=white,
                    font=\sffamily\scriptsize, text width=5.0cm, align=center,
                    inner sep=3pt},
  decision/.style ={diamond, draw=black, very thick, aspect=1.8, inner sep=1pt,
                    fill=yellow!15, font=\sffamily\scriptsize\bfseries,
                    text width=1.9cm, align=center},
  endact/.style   ={circle, draw=black, thick, fill=white, minimum size=6mm,
                    inner sep=0pt, path picture={
                      \draw[fill=black] (path picture bounding box.center)
                        circle (1.4mm);}},
  %------ swimlane wrappers ------
  regLayer/.style ={rectangle, draw=black, dashed, inner sep=14pt,
                    rounded corners=4pt, fill=blue!4},
  castLayer/.style={rectangle, draw=black, dashed, inner sep=14pt,
                    rounded corners=4pt, fill=yellow!5},
  phaselabel/.style={font=\sffamily\bfseries\small, fill=white, inner sep=3pt,
                     anchor=south west},
  %------ arrows ------
  flow/.style     ={->, thick, >={Stealth[length=2.0mm,width=1.4mm]}},
  branchflow/.style={->, thick, dashed, >={Stealth[length=2.0mm,width=1.4mm]}},
]

% =========================================================================
%  Phase B: Voter Registration (left swimlane, x = 0)
% =========================================================================
\node[start]    (bs)  at (0, 13.0) {};
\node[activity] (b1)  at (0, 11.7) {Submit voter ID, fingerprint,\\(optional) duress signal $(\tau, v_{\sigma})$};
\node[activity] (b2)  at (0, 10.2) {SHA-3 fingerprint hash\\+ liveness detection};
\node[decision] (bd1) at (0,  8.6) {Liveness\\pass?};
\node[activity] (b3)  at (0,  7.0) {Merkle eligibility check\\(admin-managed voter roster)};
\node[decision] (bd2) at (0,  5.4) {Eligible?};
\node[activity] (b4)  at (0,  3.8) {Generate ring-signature\\keypair $(\mathrm{sk}, \mathrm{pk})$};
\node[activity] (b5)  at (0,  2.0)
   {Issue SMDC credential ($k{=}5$ slots):\\
    \scriptsize $\text{idx}_{\text{real}} = \text{HMAC}(k_{\text{srv}}, \text{``smdc-real-index:''} \,\Vert\, \text{vid} \,\Vert\, \text{eid}) \bmod k$\\
    \scriptsize $\{C_i = g^{w_i} h^{r_i}\}_{i=0}^{k-1}$, $\{\pi^{\text{bin}}_i\}$, $\pi^{\text{sum}}\!:\sum w_i = 1$};
\node[activity] (b6)  at (0,  0.2)
   {(Optional) register duress-signal hash\\
    \scriptsize $h_{\text{stored}} = \text{HMAC-SHA256}(K_d, \text{signalType}\,\Vert\,\text{``:''}\,\Vert\,\text{signalValue})$};
\node[activity] (b7)  at (0, -1.3)
   {Persist voter record (SQLite):\\\scriptsize $\langle\text{vid}, \mathrm{pk}, \{C_i,\pi^{\text{bin}}_i\}, \pi^{\text{sum}}, \text{fpHash}, h_{\text{stored}}\rangle$};
\node[endact]   (be)  at (0, -2.7) {};

% Reject end-states for the two Phase-B decisions
\node[endact] (be1) at (-4.0, 8.6) {};
\node[endact] (be2) at (-4.0, 5.4) {};

% =========================================================================
%  Phase C: Vote Casting (right swimlane, x = 9.5)
% =========================================================================
\node[start]    (cs)  at (9.5, 13.0) {};
\node[activity] (c1)  at (9.5, 11.7) {Authenticate (JWT); select\\candidate $j$, SMDC slot $s_{\text{idx}}$};
\node[activity] (c2)  at (9.5, 10.2) {(Optional) submit detected\\duress signal $(\tau, v_{\sigma})$};
\node[decision] (cd1) at (9.5,  8.6) {Election\\active \&\\not voted?};
\node[activity] (c3)  at (9.5,  7.0)
   {1-hot Paillier encrypt:\\$\{E_j = \text{Enc}(\mathrm{pk}_p, v_j)\}_{j=1}^{m}$};
\node[activity] (c4)  at (9.5,  5.4)
   {CDS OR-proofs over $\{E_j\}$:\\$\{\pi^{\text{bin}}_j\}_{j=1}^m,\ \pi^{\text{sum}}\!:\sum v_j = 1$};
\node[activity] (c5)  at (9.5,  3.8)
   {$\text{bW} \!\gets\! \textsc{DuressCheck}(\tau, v_{\sigma}, h_{\text{stored}})$;\\
    $\text{fW} = \text{slot.Weight} \times \text{bW}$;\\
    $\forall j\!:\, E_{w,j} \gets E_j^{\text{fW}}$};
\node[decision] (cd2) at (9.5,  2.0) {$\text{bW}\!=\!0$?};

% Coerced branch (right of cd2)
\node[activity] (c6a) at (14.5, 2.0)
   {\textbf{Silent discard}\\\scriptsize emit audit event;\\\scriptsize do NOT mark key image;\\\scriptsize return plausible receipt};

% Genuine branch (continues down)
\node[activity] (c6b) at (9.5,  0.2)
   {Ring sign $\sigma$ + extract key image $I$;\\
    SA$^2$ split $\{S_{A,j}, S_{B,j}\}$ + Merkle proof;\\
    KeyImageStore.MarkUsed$(I)$;\\
    Submit to Fabric chaincode \texttt{CastVote}};

\node[activity] (c7)  at (9.5, -1.5)
   {Return receipt $\langle \text{vid}, \text{txID}, I \rangle$ to voter\\
    \scriptsize \emph{indistinguishable across genuine \& coerced branches}};
\node[endact]   (ce)  at (9.5, -2.9) {};

% Reject end-state for Phase C (election inactive / already voted)
\node[endact] (ce-rej) at (5.5, 8.6) {};

% =========================================================================
%  Swimlane wrappers (pastel layered, dashed border)
% =========================================================================
\begin{scope}[on background layer]
  \node[regLayer, fit=(bs)(b1)(b7)(be)(be1)(be2)] (regLyr) {};
\end{scope}
\node[phaselabel] at ([xshift=10pt,yshift=8pt]regLyr.north west)
   {Phase B: Voter Registration};

\begin{scope}[on background layer]

\end{scope}
\node[phaselabel] at ([xshift=10pt,yshift=8pt]castLyr.north west)
   {Phase C: Vote Casting};

\draw[flow] (c5) -- (cd2);
