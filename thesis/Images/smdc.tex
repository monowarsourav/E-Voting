                    minimum width=5.0cm, minimum height=1.0cm, fill=blue!8,

  credlayer/.style ={rectangle, draw=black, dashed, inner sep=10pt,
                    rounded corners=3pt, fill=blue!4},
  sumlayer/.style  ={rectangle, draw=black, thick,

   {SMDC Deniable Credential ($k=5$ slots)};

                     3/{$w_1=0$ (fake)\\$C_1=g^0\!\cdot\!h^{r_1}$\\$\pi_1^{\text{bin}}$},

  \node[credlayer, fit=(sl0)(sl4)] (cred) {};

   {Credential layer (Pedersen commitments + binary proofs)};

% =====================================================================
%  Sum proof bar (gap of ~0.65cm from slot bottom y=2.1)
%  bar at y=1.0 with height 0.85 => spans [0.575, 1.425]

\node[sumlayer] (sumproof) at (6.0, 1.0)
   {Sum proof $\pi^{\text{sum}}$: $\textstyle\sum_i w_i = 1$
    (Schnorr-style on $\prod_i C_i / g$; no information about which slot is real)};

% =====================================================================
%  Outcome branches (BOTH submitted to tally; difference is contribution)
%  branches at y=-0.7 => spans [-1.2, -0.2], gap from sumproof bottom 0.575 = 0.625cm
% =====================================================================
\node[realpath] (realbranch) at ( 2.5, -0.7)
   {\textbf{Real-slot path}\\\scriptsize voter casts on slot $i = \text{idx}_{\text{real}}$;
    $w_i=1$ multiplies into ciphertext};
\node[fakepath] (fakebranch) at (10.5, -0.7)
   {\textbf{Fake-slot path (decoy)}\\\scriptsize voter casts on any $j \neq \text{idx}_{\text{real}}$;
    $w_j=0$ multiplies into ciphertext};

\draw[flow] (sumproof.south -| realbranch) -- (realbranch.north);
\draw[flow] (sumproof.south -| fakebranch) -- (fakebranch.north);

% =====================================================================
%  Tally row: BOTH branches submit; difference is what gets counted
%  at y=-2.4 => spans [-2.9, -1.9], gap from branch bottom -1.2 = 0.7cm
% =====================================================================
\node[realpath, fill=green!5] (realtally)  at ( 2.5, -2.4)
   {\scriptsize Submitted to chain $\Rightarrow$\\\scriptsize tally adds $w=1$ contribution};
\node[fakepath, fill=blue!5] (faketally) at (10.5, -2.4)
   {\scriptsize Submitted to chain $\Rightarrow$\\\scriptsize tally adds $w=0$ (silent decoy)};

\draw[flow] (realbranch.south) -- (realtally.north);
\draw[flow] (fakebranch.south) -- (faketally.north);

% =====================================================================
%  Pedersen-hiding note (full-width, below tally row, no overlap)

   {Pedersen perfect hiding $\Rightarrow$ commitments $C_j$ for $w_j{=}0$ and $w_j{=}1$
