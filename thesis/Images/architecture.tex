% =====================================================================

\draw[flow] (voter.south) -- node[right,font=\sffamily\scriptsize,pos=0.55,fill=white,inner sep=1pt]

% =========================================================================

%  Row 3 : Privacy Layer (SA^2 two non-colluding aggregators)
% =========================================================================

  \node[layer,fit=(saA)(saB)] (privacy) {};
\end{scope}
\node[layertitle] at ([xshift=8pt,yshift=8pt]privacy.north west) {Privacy Layer (SA\textsuperscript{2})};

\draw[flow] (cryptolayer.south -| saA) --

            node[right,pos=0.3,font=\sffamily\scriptsize,fill=white,inner sep=1pt] {share$_B=E(-m)$} (saB.north);
